/usr/bin/qemu-system-x86_64 \
        -kernel linux-5.4/arch/x86/boot/bzImage \  #kernel bootable image
        -initrd $PWD/initramfs.cpio.gz \   #file system
        -fsdev local,security_model=passthrough,id=fsdev0,path=$HOME \ #shared folder
        -device virtio-9p-pci,id=fs0,fsdev=fsdev0,mount_tag=hostshare \  #shared folder
        -nographic \  #terminal 
        -monitor none \
        -s \   #debug port
        -append "console=ttyS0 nokaslr"  #nokaslr
